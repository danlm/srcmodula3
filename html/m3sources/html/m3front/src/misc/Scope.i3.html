INTERFACE Scope;

IMPORT M3, M3ID, M3Buf, Value, CG;

TYPE
  T = M3.Scope;

VAR
  Initial: T;

PROCEDURE PopNew ();

                   module := FALSE;  nested := FALSE): T;

PROCEDURE Pop  (old: T);
PROCEDURE Top  (): T;

PROCEDURE Initialize ();
PROCEDURE Reset ();
