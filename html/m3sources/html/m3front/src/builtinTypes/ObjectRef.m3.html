MODULE ObjectRef;

IMPORT M3ID, ObjectType, Tipe, Scope;

PROCEDURE Initialize () =
  VAR s: Scope.T;
  BEGIN
    s := Scope.PushNew (FALSE, M3ID.NoID);
    Scope.PopNew ();
    T := NIL;   (* the value is used by ObjectType.New ! *)
    T := ObjectType.New (NIL, TRUE, NIL, s, s);
    Tipe.Define ("ROOT", T, TRUE);
  END Initialize;

BEGIN
END ObjectRef.
