MODULE Reff;

IMPORT RefType, Tipe, M3String, TextExpr;

PROCEDURE Initialize () =
  BEGIN
    T := RefType.New (NIL, TRUE, TextExpr.New (M3String.Add ("$refany$")));
    Tipe.Define ("REFANY", T, TRUE);
  END Initialize;

BEGIN
END Reff.
