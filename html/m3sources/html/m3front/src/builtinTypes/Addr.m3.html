MODULE Addr;

IMPORT M3String, RefType, Tipe, TextExpr;

PROCEDURE Initialize () =
  BEGIN
    T := RefType.New (NIL, FALSE, TextExpr.New (M3String.Add ("$address$")));
    Tipe.Define ("ADDRESS", T, TRUE);
  END Initialize;

BEGIN
END Addr.
