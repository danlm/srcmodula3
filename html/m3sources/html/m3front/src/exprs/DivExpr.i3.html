INTERFACE DivExpr;

IMPORT Expr, Target;

PROCEDURE New (a, b: Expr.T): Expr.T;

PROCEDURE SmallPowerOfTwo (READONLY x: Target.Int;  VAR log: INTEGER): BOOLEAN;
If 'x' is a small positive power of 2, sets 'log' to the base 2 logorithm of 'x' and returns 'TRUE', otherwise returns 'FALSE' 

END DivExpr.
