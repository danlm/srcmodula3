MODULE CoffTime;

IMPORT Time, File, FS, OSError;

PROCEDURE Now (): INTEGER =
  VAR now := Time.Now ();
  BEGIN
    RETURN ROUND (now - EpochAdjust);
  END Now;

PROCEDURE OfFile (file: TEXT): INTEGER =
  VAR s: File.Status;
  BEGIN
    TRY
      s := FS.Status (file);
    EXCEPT OSError.E =>
      RETURN 0;
    END;
    RETURN ROUND (s.modificationTime - EpochAdjust);
  END OfFile;

BEGIN
END CoffTime.
