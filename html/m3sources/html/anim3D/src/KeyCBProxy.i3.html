INTERFACE KeyCBProxy;

IMPORT CB, ProxiedObj;

FROM KeyCB IMPORT T, Rec;

   registered by assigning it to MkProxyT. 


VAR
  MkProxyT : PROCEDURE (x : T) := NIL;

TYPE
  Proxy = ProxiedObj.Proxy OBJECT
  METHODS
    invoke (kr : Rec) RAISES {CB.BadMethod};
  END;

END KeyCBProxy.
