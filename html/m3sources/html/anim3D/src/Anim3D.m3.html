MODULE Anim3D;

IMPORT AnimServer, Clock, Wr;

VAR
  clock: Clock.T;

PROCEDURE Now () : LONGREAL =
  BEGIN
    IF clock = NIL THEN
      clock := NEW (Clock.T).init ();
    END;
    RETURN clock.time ();
  END Now;

PROCEDURE ChangeClock (c : Clock.T) =
  BEGIN
    clock := c;
  END ChangeClock;

PROCEDURE SetErrorWr (wr : Wr.T) =
  BEGIN
    AnimServer.SetErrorWr (wr);
  END SetErrorWr;

BEGIN
  lock := AnimServer.externalLock;
END Anim3D.
