INTERFACE GroupGOPrivate;

IMPORT GO;

FROM GroupGO IMPORT T, Public;

REVEAL
  T <: Private;

TYPE
  Private = Public BRANDED OBJECT
    children : REF ARRAY OF GO.T;
    last     : INTEGER;   (* children[0 .. last] contains good data *)
  END;

END GroupGOPrivate.
