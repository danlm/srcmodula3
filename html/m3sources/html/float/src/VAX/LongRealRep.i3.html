INTERFACE LongRealRep;
This interface describes the layout of VAX double precision reals. 

TYPE
  T = RECORD
    fraction0 : BITS  7 FOR [0..63]         := 0;
    exponent  : BITS  8 FOR [0..255]        := 0;
    sign      : BITS  1 FOR [0..1]          := 0;
    fraction1 : BITS 16 FOR [0..64*1024-1]  := 0;
    fraction2 : BITS 16 FOR [0..64*1024-1]  := 0;
    fraction3 : BITS 16 FOR [0..64*1024-1]  := 0;
  END;

CONST
  ExponentBias = 128;

CONST
  NegInf = T { sign := 1, exponent := 255 };
  PosInf = T { sign := 0, exponent := 255 };
  Nan    = T { sign := 1, exponent := 0   };

END LongRealRep.
