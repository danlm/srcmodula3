INTERFACE Wrx86;

IMPORT Wr, Target;

FROM M3CG IMPORT Name, TypeUID;
FROM M3CG IMPORT Var, Proc, Label;
FROM M3CG IMPORT Type;

TYPE T <: Public;
TYPE Public = OBJECT
      METHODS
        Flush ();
        NL ();
        Cmd (cmd: TEXT);
        ZName (n: Name);
        VName (v: Var);
        PName (p: Proc);
        TName (t: Type);
        Flt (READONLY f: Target.Float);
        Bool (b: BOOLEAN);
        Lab (i: Label);
        Tipe (t: TypeUID);
        Int (i: INTEGER);
        TInt (READONLY i: Target.Int);
        BInt (i: INTEGER);
        Txt (t: TEXT);
        OutC (c: CHAR);
        OutT (txt: TEXT);
        OutN (n: Name);
        OutS (READONLY buf: ARRAY OF CHAR);
        OutI (i: INTEGER);
      END;

PROCEDURE New (wr: Wr.T): T;

END Wrx86.
