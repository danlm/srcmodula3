MODULE FSError;

IMPORT OSError, OSErrorWin32, WinError;

PROCEDURE FileNotFound (err: OSError.Code): BOOLEAN =
  BEGIN
    RETURN err.head = OSErrorWin32.ErrnoAtom (WinError.ERROR_FILE_NOT_FOUND)
  END FileNotFound;

BEGIN
END FSError.
