MODULE M3AST_AS_Init;

IMPORT AST, M3AST_AS, M3AST_AS_F;

PROCEDURE Exp_used_id(n: M3AST_AS_F.Exp_used_id): AST.NODE RAISES {}=

    n.vUSED_ID := NEW(M3AST_AS.USED_ID).init();
