IMPORT Err;

PROCEDURE Check(assertion: BOOLEAN) RAISES {} =
  BEGIN
    IF NOT assertion THEN Fail() END;
  END Check;

PROCEDURE Fail() RAISES {} =
  BEGIN
    Err.Print("Assertion failed", Err.Severity.Disaster);
  END Fail;

BEGIN

END M3Assert.
