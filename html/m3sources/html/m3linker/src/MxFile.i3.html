INTERFACE MxFile;

IMPORT File, OSError;

PROCEDURE Read (f: File.T; VAR(*OUT*)buf: ARRAY OF CHAR; len: INTEGER): INTEGER
  RAISES {OSError.E};

 == f.read (LOOPHOLE(buf[0..len-1], ARRAY OF BYTE), mayBlock := TRUE) 

END MxFile.
