MODULE WinAux;

IMPORT VBT, VBTClass, WinDef, WinTrestle;

PROCEDURE WindowHandle (v: VBT.T): WinDef.HWND =
  BEGIN
    WHILE v # NIL DO
      TYPECASE v.upRef OF
      | WinTrestle.Child (ch) =>
          IF ch # NIL THEN
            RETURN ch.hwnd
          END
      ELSE
        (* skip *)
      END;
      v := v.parent;
    END;
    RETURN NIL;
  END WindowHandle;

BEGIN
END WinAux.
