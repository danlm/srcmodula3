MODULE TrestleOS;

IMPORT WinTrestle;

VAR mu := NEW(MUTEX);
    inited := FALSE;

PROCEDURE Init () =
  BEGIN
    LOCK mu DO
      IF NOT inited THEN
        WinTrestle.Init();
        inited := TRUE
      END
    END
  END Init;

PROCEDURE UserName (): TEXT =
  BEGIN
    RETURN "Unknown user"
  END UserName;

BEGIN
END TrestleOS.
