UNSAFE MODULE Params;

IMPORT RTArgs;

EXCEPTION FatalError; <* FATAL FatalError *>

PROCEDURE Get(n: CARDINAL): TEXT =
  BEGIN
    IF n >= Count THEN RAISE FatalError END;
    RETURN RTArgs.GetArg(n);
  END Get;

BEGIN
  Count := RTArgs.ArgC();
END Params.
