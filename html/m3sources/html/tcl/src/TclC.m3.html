UNSAFE MODULE TclC;

IMPORT Cstdlib;

PROCEDURE FreeResult (interp: Interp_star) =
  BEGIN
    IF interp.freeProc # NIL THEN
      interp.freeProc (interp.result);
      interp.freeProc := NIL;
    END;
  END FreeResult;

BEGIN
  volatile := LOOPHOLE (-1, FreeProc);
  static   := LOOPHOLE (0, FreeProc);
  dynamic  := LOOPHOLE (Cstdlib.free, FreeProc);
END TclC.
