A Char.T is an CHAR. This interface is intended to be used as an actual interface for generic interfaces and modules such as Table. 

INTERFACE Char;

IMPORT Word;

TYPE T = CHAR;

CONST Brand = "Char";

PROCEDURE Compare (a, b: T): [-1..1];
== RETURN (a - b) 

PROCEDURE Equal (a, b: T): BOOLEAN;
== RETURN (a = b) 

PROCEDURE Hash (a: T): Word.T;
== RETURN ORD (a) 

END Char.
